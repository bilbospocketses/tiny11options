/// <summary>

/// Mirrors the existing logic in HeadlessRunner.QuoteIfNeeded but factored
/// so BuildHandlers/CleanupHandlers/DismountSourceIsoIfApplicable can use
/// the same escaping discipline. v1.0.8 audit WARNING launcher B3.

public static class ArgQuoting

function Build-RelaunchArgs {
    [CmdletBinding()]
    param([Parameter(Mandatory)]$Bound, [Parameter(Mandatory)][string]$ScriptPath)
    $parts = @("-NoProfile", "-File", "`"$ScriptPath`"")
    foreach ($entry in $Bound.GetEnumerator()) {
        if ($entry.Key -eq 'Internal') { continue }
        $val = $entry.Value
        if ($val -is [switch]) {
            if ($val.IsPresent) { $parts += "-$($entry.Key)" }
        } elseif ($val -is [array]) {
            # v1.0.8 audit WARNING scripts A2: repeat the param name per array element
            foreach ($item in $val) {
                $parts += "-$($entry.Key)"
                $parts += "`"$($item -replace '"','""')`""
            }
        } else {
            # v1.0.8 audit WARNING scripts A2: escape embedded " in scalar values
            $parts += "-$($entry.Key)"
            $parts += "`"$($val -replace '"','""')`""
        }
    }
    $parts -join ' '
}
